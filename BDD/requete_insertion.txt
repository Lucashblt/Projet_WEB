-- Insérer la déclinaison du produit
INSERT INTO declinaisonproduit(idProduit, idTaille, idCouleur)
SELECT td_produit.idproduit, td_taille.idtaille, td_couleur.idcouleur
FROM 
(SELECT idProduit FROM produit WHERE nom = 'Puma Suede') as td_produit
  CROSS JOIN      (SELECT idTaille FROM tailleproduit WHERE idType = (SELECT idType FROM typeproduit WHERE nom = 'Sneakers')) as td_taille
  CROSS JOIN     (SELECT idCouleur FROM couleurproduit WHERE nom = 'Blanc' OR nom='Rouge' OR nom='noir' OR nom='Bleu' ) as td_couleur ;

-- Insérer le prix du produit
INSERT INTO prixproduit (idProduit, dateDebut, prixBrut, prixNet)

        from declinaisonproduit 
        WHERE idProduit = (SELECT idProduit FROM produit WHERE nom = 'Puma Suede');
